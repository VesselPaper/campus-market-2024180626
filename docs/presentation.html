// 路由配置（部分）
{ path: '/', name: 'home', component: HomeView }
{ path: '/market', name: 'market-list', component: MarketListView }
{ path: '/item/:id', name: 'item-detail', component: ItemDetailView }
{ path: '/publish', name: 'publish', component: PublishView }
{ path: '/login', name: 'login', component: LoginView }
{ path: '/:pathMatch(.*)*', name: 'not-found', component: NotFoundView }

// 导航守卫：未登录跳转登录页
router.beforeEach(async (to) => {
  if (to.name !== 'login' && ...) {
    await userStore.restoreSession()
    if (!userStore.currentUser)
      return { name: 'login' }
  }
})

🎯 设计要点

命名路由 — 每个路由设置 name，导航时用 name 代替 path，避免硬编码 URL

🔀 动态路由

/item/:id 通过 URL 参数传递商品 ID，详情页根据 ID 加载对应数据

🛡️ 导航守卫

每次路由跳转前检查登录态，未登录用户自动跳转到身份创建页

🔄 路由重定向

/trade → /market?type=secondhand 等 4 个重定向，简化路由结构

❌ 404 兜底

/:pathMatch(.*)* 匹配所有未定义路径，显示 NotFound 页面

📌 核心代码 · Pinia 状态管理

src/stores/ — 4 个 Store 驱动全应用状态

// userStore.ts — 用户状态管理
export const useUserStore = defineStore('user', () => {
  const currentUser = ref<User | null>(null)

  // 从 localStorage 恢复登录
  async function restoreSession() {
    const id = localStorage.getItem('campus_market_user_id')
    if (!id) return false
    const found = users.value.find(
      u => u.id === Number(id)
    )
    if (found) { currentUser.value = found; return true }
    return false
  }

  function clearUser() {
    currentUser.value = null
    localStorage.removeItem('campus_market_user_id')
  }

  return { currentUser, restoreSession, clearUser, ... }
})

🎯 为什么需要 Store？

收藏状态需要在列表页、详情页、个人中心三处同步。如果没有 Store，每个页面各自管理数据，收藏后其他页面不知道状态变了。

🔄 状态同步原理

所有页面读写同一个 Store 实例。Pinia 基于 Vue 的响应式系统，Store 数据变化时，所有引用了该数据的组件自动重新渲染。

📋 4 个 Store 职责

userStore — 登录/注册/会话恢复/退出
itemStore — 商品 CRUD/api 调用封装
favoriteStore — 收藏切换/favoriteCount 同步
messageStore — 会话管理/消息发送/未读计数

📌 核心代码 · 收藏状态同步

favoriteStore.ts — 一处修改，处处更新

// toggleFavorite — 收藏切换核心逻辑
async function toggleFavorite(itemId: number): Promise<boolean> {
  const existing = await getFavoriteRecord(itemId)
  if (existing) {
    // 已收藏 → 取消
    await removeFavorite(existing.id)
    favorites.value = favorites.value.filter(
      f => f.id !== existing.id
    )
  } else {
    // 未收藏 → 添加
    const res = await addFavorite({ userId, itemId, ... })
    favorites.value.push(res.data)
  }
  // 同步更新 favoriteCount
  const actualCount = await countFavorites(itemId)
  useItemStore().editItem(itemId, { favoriteCount: actualCount })
  return !existing
}

🎯 乐观更新

点击收藏 → 前端立即更新 UI → 同时异步调用 API。用户获得即时反馈，无需等待网络请求完成。

🔄 跨页面同步

修改 Store 中的 favorites 数组 → 所有页面的收藏列表自动更新。取消收藏后，列表页的收藏按钮立刻恢复为"未收藏"。

🔗 favoriteCount 联动

切换收藏后重新查询实际收藏数，同步更新到 itemStore 中的 favoriteCount 字段。商品卡片上的 ♥ 数字实时反映真实收藏人数。

📌 核心代码 · 组件通信模式

Props 向下传数据，Emit 向上发事件，Store 全局共享

⬇️

Props 父→子

<!-- 父组件 -->
<MarketItemCard
  :item="item"
  :visited="visited"
/>

<!-- 子组件接收 -->
defineProps<{
  item: Item
  visited?: boolean
}>()
⬆️

Emit 子→父

// 子组件发射事件
const emit = defineEmits<{
  click: [id: number]
}>()
emit('click', item.id)

// 父组件监听
<MarketItemCard
  @click="goToDetail"
/>
🔄

Store 全局共享

// 任何组件都可以读写
const userStore = useUserStore()
const favoriteStore = useFavoriteStore()

// 数据变化自动同步到
// 所有引用了该数据的组件
userStore.currentUser
favoriteStore.favorites
